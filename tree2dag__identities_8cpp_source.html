31 // Search in tree n2 there is a subtree equal to n1, which is not n1
32 void FindAndLinkIdentiesRec(BinaryDrag<conact>::node* n1, BinaryDrag<conact>::node* n2) {
33  if (n2->isleaf() || n1 == n2)
34  return;
35 
36  if (n1 != n2->left && EqualTrees(n1, n2->left)) {
37  n2->left = n1;
38  }
39 
40  if (n1 != n2->right && EqualTrees(n1, n2->right)) {
41  n2->right = n1;
42  }
43 
44  FindAndLinkIdentiesRec(n1, n2->left);
45  FindAndLinkIdentiesRec(n1, n2->right);
46 }

48 void Tree2DagUsingIdentitiesRec(BinaryDrag<conact>::node *n, BinaryDrag<conact>& t) {
49  FindAndLinkIdentiesRec(n, t.GetRoot());
